\begin{abstract}
Este documento estudia los determinantes y la capacidad predictiva de la variabilidad extrema del tipo de cambio mediante percentiles de retornos condicionales. Se comparan metodos de aprendizaje de maquina con enfoques basados en Bayesian Additive Regression Trees (BART), enfatizando desempeno en cola izquierda y cola derecha de la distribucion de retornos.
\end{abstract}

\textbf{Palabras clave:} tipo de cambio, riesgo extremo, quantiles, machine learning, BART, tail risk.

\textbf{Clasificacion JEL:} C53, F31, G17.

\section{Datos y variables}
\subsection{Variable objetivo}
Defina retornos logaritmicos como:
\begin{equation}
    r_{t+h} = 100\times\left(\log S_{t+h} - \log S_t\right),
\end{equation}
donde $S_t$ es el nivel del tipo de cambio nominal.

\subsection{Medidas de variabilidad extrema}
Para cada horizonte $h$, estime quantiles condicionales:
\begin{equation}
    Q_{\tau}(r_{t+h}\mid X_t),\quad \tau \in \{0.01,0.05,0.10,0.90,0.95,0.99\}.
\end{equation}

\subsection{Metricas de evaluacion}
Use funcion de perdida quantilica (pinball loss):
\begin{equation}
    L_{\tau}(u)=u\left(\tau-\mathbb{I}\{u<0\}\right),\quad u=r_{t+h}-\hat{Q}_{\tau,t+h}.
\end{equation}

Compare modelos con:
\begin{itemize}[leftmargin=*]
    \item Pinball loss promedio por $\tau$ y horizonte $h$.
    \item Cobertura empirica de eventos extremos.
    \item Tests de comparacion predictiva (por ejemplo, Diebold-Mariano).
\end{itemize}
